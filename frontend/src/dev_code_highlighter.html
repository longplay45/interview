def calculate_mode(data):
    mode_count = 0
    mode = None
    
    for value in data:
        count = data.count(value)
        if count > mode_count:
            mode_count = count
            mode = value
    
    return mode

data = [1, 2, 2, 3, 3, 3, 4, 4, 5]
mode = calculate_mode(data)
print(mode)
